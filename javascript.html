            C = A % B it stores reminder of division of A by B, C = 4

            entries()   -> object to array

            }

            let arr1 = [["p_id", 111],["p_name", "p_one"],["p_cost", 10000]]

            console.log(obj1)   //{p_id: 111, p_name: 'p_one', p_cost: 10000}
